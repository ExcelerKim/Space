def perfectTest(i):
    k = []
    flag = True
    for j in range(1,i):
        if i%j == 0:
            k.append(j)
        else:
            pass
    if sum(k) == i:
        flag = True
    else:
        flag = not flag
    return flag

def writing(t_list):
    text = ''
    i = 0
    n = len(t_list)
    for t in t_list:
        if i<n-1:
            text += t+', '
            i +=1
        else:
            text +='and '+t_list[-1]
    return text
